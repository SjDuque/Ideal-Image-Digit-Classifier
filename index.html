#This method takes in a file name and loads it up into a numpy array.
#is_img indicates if the file is for images, which makes a np array of shape (num_ex x num_image_rows x num_image_cols)
#the second numpy array is the labels, which is num_ex x 1
def mnist_to_np(f_name, is_img = False):
    with gzip.open(f_name) as mnist:
        #for information about loading up mnist refer to http://yann.lecun.com/exdb/mnist/
        magic_num = int.from_bytes(mnist.read(4), byteorder='big', signed=False) #first int is the magic number
        num_ex = int.from_bytes(mnist.read(4), byteorder='big', signed=False) #second int is the number of examples
        if is_img == True:
            rows = int.from_bytes(mnist.read(4), byteorder='big', signed=False) #if it is an image then the 3rd integer is for num_rows
            cols = int.from_bytes(mnist.read(4), byteorder='big', signed=False) #if it is an image then the 4th integer is for num_columns
        if is_img == True:
            data = np.array(list(mnist.read(num_ex * rows * cols))).reshape(num_ex, rows, cols) #reshapes the data to the appropriate image shape
        else:
            data = np.array(list(mnist.read(num_ex))).reshape(num_ex, 1) #this just reads the examples as 
        return data

Ideal Images¶

This function converts an examples in an image matrix to a single image. This is what will be the "ideal image". The ideal_images method organizes each image class into the corresponding index in the ideal_images_list.

The last block converts the training data into the ideal images and stores the labels into a variable named labels.

#then multiplies each image by 255.
def norm_sum_images(images):
    raw_sum = np.mean(images, axis = 0)
    #norm_sum = (raw_sum / np.max(raw_sum))*255
    norm_sum = raw_sum
    return norm_sum

def ideal_images(images, labels = None):
    
    if labels is None:
        labels = images[1]
        images = images[0]
        
    ideal_image_list = []
    #the for loop goes through 0 to 9 because there are 10 classes, each labeled 0-9
    for l in range(num_labels):
        l_images = images[labels[:,0] == l]
        ideal_image_list.append(norm_sum_images(l_images).astype(np.uint8))
    
    return ideal_image_list

def show_img_list(image_list):
    num_img = len(image_list)
    
    rows = math.ceil(math.sqrt(num_img))
    cols = math.floor(math.sqrt(num_img))
    fig = plt.figure(figsize = (10, 10))
    
    for i in range(0, num_img):
        fig.add_subplot(rows, cols, i+1)
        img = image_list[i].astype(np.uint8)
        show_img(img)

Prediction making¶

The following methods are made to calculate predictions of images based on the similarity to the ideal images. Similar to k-nearest neighbor, but only comparing it to the ideal images.

In [11]:
#This method calculates the difference between the image and ideal image.
#You are welcome to mess around with this method to see if you can find a
#better way to compare.
def square_dif(image, ideal_image):
    
    dif = (image-ideal_image).astype(np.float32)
    
    return (dif ** 4)/(255**3)
In [12]:
#This method determines how close the images are and assigns a
#score for each example
def score(image, ideal_image):
    dif = square_dif(image, ideal_image).reshape(image.shape[0], -1)
    #the reshape made the image array into shape (num_ex, rows*cols)
    dif = np.sqrt(np.sum(dif, axis = 1, keepdims = True)) #sum along the image axis
    
    return 1/(dif)

def score_labels(images, ideal_image_list):
    scores = []
    for l in range(num_labels):
         #calculates the score compared to this label's ideal image
        scores.append(score(images, ideal_image_list[l]).reshape(-1))
    return np.array(scores)

def determine_labels(images, ideal_image_list):
    scores = score_labels(images, ideal_image_list)
                          
    #this finds the index of the max element along the label-axis                      
    max_scores = scores.argmax(axis=0)
    
    return max_scores

def accuracy(actual_labels, predicted_labels):
    correct = actual_labels.reshape(-1) == predicted_labels.reshape(-1)
    
    return np.sum(correct) / actual_labels.shape[0]
